00066   Math::real Geodesic::SinSeries(real sinx, real cosx,
00067                                  const real c[], int n) throw() {
00068     // Evaluate y = sum(c[i] * sin(2 * i * x), i, 1, n) using Clenshaw
00069     // summation.  (c[0] is unused.)
00070     // Approx operation count = (n + 5) mult and (2 * n + 2) add
00071     real
00072       ar = 2 * (cosx - sinx) * (cosx + sinx), // 2 * cos(2 * x)
00073       y0 = n & 1 ? c[n--] : 0, y1 = 0;        // Accumulators for sum
00074     // Now n is even
00075     while (n) {
00076       // Unroll loop x 2, so accumulators return to their original role
00077       y1 = ar * y0 - y1 + c[n--];
00078       y0 = ar * y1 - y0 + c[n--];
00079     }
00080     return 2 * sinx * cosx * y0; // sin(2 * x) * y0
00081   }

00317   Math::real Geodesic::Astroid(real x, real y) throw() {
00318     // Solve k^4+2*k^3-(x^2+y^2-1)*k^2-2*y^2*k-y^2 = 0 for positive root k.
00319     // This solution is adapted from Geocentric::Reverse.
00320     real k;
00321     real
00322       p = sq(x),
00323       q = sq(y),
00324       r = (p + q - 1) / 6;
00325     if ( !(q == 0 && r <= 0) ) {
00326       real
00327         // Avoid possible division by zero when r = 0 by multiplying equations
00328         // for s and t by r^3 and r, resp.
00329         S = p * q / 4,            // S = r^3 * s
00330         r2 = sq(r),
00331         r3 = r * r2,
00332         // The discrimant of the quadratic equation for T3.  This is zero on
00333         // the evolute curve p^(1/3)+q^(1/3) = 1
00334         disc =  S * (S + 2 * r3);
00335       real u = r;
00336       if (disc >= 0) {
00337         real T3 = S + r3;
00338         // Pick the sign on the sqrt to maximize abs(T3).  This minimizes loss
00339         // of precision due to cancellation.  The result is unchanged because
00340         // of the way the T is used in definition of u.
00341         T3 += T3 < 0 ? -sqrt(disc) : sqrt(disc); // T3 = (r * t)^3
00342         // N.B. cbrt always returns the real root.  cbrt(-8) = -2.
00343         real T = Math::cbrt(T3); // T = r * t
00344         // T can be zero; but then r2 / T -> 0.
00345         u += T + (T != 0 ? r2 / T : 0);
00346       } else {
00347         // T is complex, but the way u is defined the result is real.
00348         real ang = atan2(sqrt(-disc), -(S + r3));
00349         // There are three possible cube roots.  We choose the root which
00350         // avoids cancellation.  Note that disc < 0 implies that r < 0.
00351         u += 2 * r * cos(ang / 3);
00352       }
00353       real
00354         v = sqrt(sq(u) + q),    // guaranteed positive
00355         // Avoid loss of accuracy when u < 0.
00356         uv = u < 0 ? q / (v - u) : u + v, // u+v, guaranteed positive
00357         w = (uv - q) / (2 * v);           // positive?
00358       // Rearrange expression for k to avoid loss of accuracy due to
00359       // subtraction.  Division by 0 not possible because uv > 0, w >= 0.
00360       k = uv / (sqrt(uv + sq(w)) + w);   // guaranteed positive
00361     } else {               // q == 0 && r <= 0
00362       // y = 0 with |x| <= 1.  Handle this case directly.
00363       // for y small, positive root is k = abs(y)/sqrt(1-x^2)
00364       k = 0;
00365     }
00366     return k;
00367   }
